spots/
  [spotId]
    // Info Básica
    name: string
    description: string
    gallery: string[]

    // Contacto
    contactEmail: string
    contactPhone: string
    contactWebsite: string
    
    // Deportes Disponibles
    availableSports: array (de strings)
    
    // Ubicación
    location: geopoint
    geohash: string
    
    // Estado y Verificación
    isActive: boolean
    isVerified: boolean
    
    // Métricas Agregadas (denormalizadas)
    overallRating: number
    reviewsCount: number 
    activeMeetupsCount: number
    
    // Metadatos
    createdAt: timestamp
    updatedAt: timestamp
    createdBy: reference (users collection)

    // ============ SUBCOLECCIONES ============

        spotId: string
        sportRatings: map

            difficulty: number
            content?: string

        reports: number

        updatedAt: timestamp
        isDeleted?: boolean

            likesCount: number

            commentsCount: number
            reports: number
            createdAt: timestamp
            updatedAt: timestamp
            isDeleted: boolean

    [SUBCOLECCIÓN]
    /meetups
      [meetupId]
        // Información Básica
        spotId: string
        organizerId: string
        chatId: string  // Referencia al chat
        title: string
        description?: string
        tags?: string[]
        
        // Tipo de Meetup
        type: 'CASUAL' | 'TOURNAMENT' | 'MATCH' | 'ROUTINE'
        
        // Estado y Visibilidad
        status: 'SCHEDULED' | 'ACTIVE' | 'CANCELLED' | 'COMPLETED'
        visibility: 'OPEN' | 'CLOSED'
        
        // Participantes (Metadata)
        participantsCount: number
        joinRequests?: string[]
        
        // Campos específicos por tipo:
        sport?: string
        minParticipants?: number
        participantLimit?: number
        date?: timestamp
        
        daysOfWeek?: number[]
        time?: string
        nextDate?: timestamp
        
        expiresAt?: timestamp
        remindersSent?: map

        // Metadatos
        createdAt: timestamp
        updatedAt: timestamp

chats/
  [chatId]
    type: 'direct' | 'group' | 'meetup'
    name?: string
    description?: string
    photoURL?: string
    createdBy?: string
    meetupId?: string
    meetupSpotId?: string

        senderId: string
        mediaUrl?: string
        mediaType?: 'image' | 'video'
        isDeleted: boolean
        createdAt: timestamp
